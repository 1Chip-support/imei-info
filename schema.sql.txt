-- Эта команда удалит старую таблицу, если она уже есть (чтобы не было ошибок)
DROP TABLE CHECKS;

-- Создаем новую таблицу
CREATE TABLE CHECKS (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, -- Уникальный номер записи
    name VARCHAR2(100) NOT NULL,                            -- Имя плательщика
    email VARCHAR2(100) NOT NULL,                           -- Email
    amount NUMBER NOT NULL,                                 -- Сумма платежа
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP          -- Дата и время создания
);

-- Проверяем, что таблица создалась (пока она пустая)
SELECT * FROM CHECKS;
